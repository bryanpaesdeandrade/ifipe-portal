export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') return res.status(200).end(); const texto = req.body.texto || req.body.lista; const API_KEY = process.env.GEMINI_API_KEY; try { // Usando a rota v1beta que é a mais compatível com o modelo flash const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: "Você é um extrator de dados Apple. Analise a lista e retorne APENAS um array JSON puro. Regras: 1. Ignore itens sem preço. 2. Campos: modelo, capacidade, cor, preco (número), quantidade (número), condicao (Novo ou CPO ou Seminovo). Lista: " + texto }] }] }) }); const data = await response.json(); if (data.error) throw new Error(data.error.message); const textoIA = data.candidates[0].content.parts[0].text; const inicio = textoIA.indexOf('['); const fim = textoIA.lastIndexOf(']'); const jsonLimpo = textoIA.substring(inicio, fim + 1); return res.status(200).json(JSON.parse(jsonLimpo)); } catch (error) { return res.status(500).json({ error: error.message }); } }
