async function cadastrarBuscador(e) { e.preventDefault(); const btn = document.getElementById('btnCadastrarBuscador'); btn.disabled = true; btn.innerText = 'PROCESSANDO...'; // Captura os dados do formulário const nome = document.getElementById('buscadorNome').value.trim().toUpperCase(); const pais = document.getElementById('buscadorPais').value; const estado = pais === 'Brasil' ? document.getElementById('buscadorEstado').value : null; const cidade = pais === 'Brasil' ? document.getElementById('buscadorCidade').value : null; const ddi = document.getElementById('buscadorDdi').value.replace('+', ''); const numero = document.getElementById('buscadorWhatsappNumero').value.replace(/\D/g, ''); const email = document.getElementById('buscadorEmail').value.trim().toLowerCase(); const senha = document.getElementById('buscadorSenha').value.trim(); // Validação básica if (!nome || !pais || !ddi || !numero || !email || !senha) { alert('Preencha todos os campos.'); btn.disabled = false; btn.innerText = 'CADASTRAR'; return; } if (pais === 'Brasil' && (!estado || !cidade)) { alert('Selecione estado e cidade.'); btn.disabled = false; btn.innerText = 'CADASTRAR'; return; } const whatsappCompleto = ddi + numero; // 1. Criar o usuário no Auth const { data: authData, error: authError } = await _s.auth.signUp({ email: email, password: senha }); if (authError) { btn.disabled = false; btn.innerText = 'CADASTRAR'; if (authError.message.includes('User already registered')) { alert('Este e-mail já está cadastrado.'); } else { alert('Erro ao criar conta: ' + authError.message); } return; } // 2. Se o Auth criou com sucesso, insere na tabela clientes_buscador if (authData?.user) { const { error: insertError } = await _s .from('clientes_buscador') .insert([{ auth_id: authData.user.id, // ID do Auth nome: nome, email: email, whatsapp: whatsappCompleto, pais: pais, estado: estado, cidade: cidade, senha: senha, // mantido por compatibilidade ativo: true, logins_count: 0 }]); if (insertError) { // Se falhar o insert, idealmente deveríamos remover o usuário do Auth, mas é complexo. // Por ora, avisamos o usuário. btn.disabled = false; btn.innerText = 'CADASTRAR'; alert('Erro ao salvar dados do perfil. Contate o suporte.'); console.error(insertError); return; } // Sucesso alert('Cadastro realizado! Verifique seu e-mail para confirmar sua conta e faça login.'); fecharModalCadastro('buscador'); } btn.disabled = false; btn.innerText = 'CADASTRAR'; }
